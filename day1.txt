# declare class
class User():
    name=''
    email=''
    #constructor
    def __init__(self,name,email):
        self.name=name
        self.email=email
    # this fun is called when user print the object
    def __str__(self):
        return(f" the username is {self.name} and email is {self.email}")

user= User('admin','admin@mail.com')
print(user)
